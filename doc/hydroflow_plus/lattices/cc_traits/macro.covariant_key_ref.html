Macro hydroflow_plus::lattices::cc_traits::covariant_key_ref

macro_rules! covariant_key_ref {
    () => { ... };
}
Expand description

Automatically defines the KeyedRef::upcast_item_ref function using the covariance of the KeyRef<'a> type with regards to 'a.

Example

use cc_traits::{Collection, Keyed, KeyedRef, covariant_key_ref};

pub struct MyMap<K, V>(std::collections::HashMap<K, V>);

impl<K, V> Collection for MyMap<K, V> {
  type Item = V;
}

impl<K, V> Keyed for MyMap<K, V> {
  type Key = K;
}

impl<K, V> KeyedRef for MyMap<K, V> {
  type KeyRef<'a>
  = &'a K where Self: 'a;

  covariant_key_ref!();
}
